<p>def crear_universo(nombre, cantidad_galaxias, cantidad_sistemas, cantidad_planetas_por_cuenta)</p>

<pre>universo = Universo.create! nombre: nombre, cantidad_galaxias: cantidad_galaxias, cantidad_sistemas: cantidad_sistemas, cantidad_planetas_por_cuenta: cantidad_planetas_por_cuenta puts universo return universo</pre>

<p>end</p>

<p>def crear_jugador(nombre)</p>

<pre>jugador = Jugador.create! nombre: nombre puts jugador return jugador</pre>

<p>end</p>

<p>task cargar: :environment do</p>

<pre>universo = crear_universo &#39;Andromeda&#39;, 9, 499, 9</pre>

<p>end</p>

<pre><code>universo = Universo.first holyman = crear_jugador &#39;Holyman&#39; gius = crear_jugador &#39;Gius&#39; sxadvi = crear_jugador &#39;Sxadvi&#39; vassili = crear_jugador &#39;Vassili&#39;

universo.cuentas.create! [

{ jugador: holyman },
{ jugador: gius },
{ jugador: sxadvi },
{ jugador: vassili }
