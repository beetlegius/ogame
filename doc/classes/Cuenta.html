def puede_atacar_a?(una_cuenta)
  return false if una_cuenta == self  
  return true if !una_cuenta.esta_protegida?

  puntos_contrincante = una_cuenta.puntos
  puntos * 0.2 < puntos_contrincante && puntos_contrincante * 0.2 < puntos
end
